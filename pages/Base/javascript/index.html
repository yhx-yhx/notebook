常见排序 ​

1.冒泡排序 ​

javascript
let array = [3, 4, 6, 7, 8, 2, 5, 4]
for (let j = 0; j < array.length - 1; j++) {
      for (let i = 0; i < array.length - 1 - j; i++) {
          if (array[i] > array[i + 1]) {
             let temp = array[i]
             array[i] = array[i + 1];
             array[i + 1] = temp
           }
       }
}
console.log(array);
